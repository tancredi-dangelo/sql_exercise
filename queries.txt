
1. SELECT * 
	FROM customers 
	WHERE name = "Mario";

2. SELECT name, surname 
	FROM customers  
	WHERE year_of_birth = 1982;

3. SELECT COUNT(*) 
	FROM invoices  
	WHERE iva = 20;

4. SELECT * 
	FROM products p 
	WHERE EXTRACT(YEAR FROM p.data) = 2017 
	AND (p.is_in production = TRUE OR p.is_on_the_market = TRUE);

5. SELECT * 
	FROM invoices i 
	JOIN customers c 
	ON i.customer_id = c.customer.id
	WHERE i.amount < 1000;

6. SELECT i.invoice_number, i.amount, i.iva, i.date, s.denomination
    FROM invoices i 
    JOIN suppliers s
    ON s.supplier_number = i.supplier_number


7. SELECT EXTRACT(YEAR FROM i.date) AS year, COUNT(*) as invoices_count
    FROM invoices i
    WHERE i.iva < 20
    GROUP BY year
    ORDER BY year

8. SELECT EXTRACT(YEAR FROM i.date) AS year, COUNT(*) AS invoices_count, SUM(i.amount) AS total_amount
    FROM invoices i
    GROUP BY year
    ORDER BY year
